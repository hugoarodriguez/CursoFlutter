/*NOTA: Las clases abstractas son utilizadas para obligar a los desarrolladores
 * a implementar las propiedades y mÃ©todos de la clase abstracta (Animal) en la clase
 * implementada (Perro)*/

void main(){
  final perro = new Perro();
  perro.emitirSonido();
  
  final gato = new Gato();
  gato.emitirSonido();
}

abstract class Animal{
  int patas;
  
  void emitirSonido();
}

class Perro implements Animal{
  int patas;
  int colas;
  
  void emitirSonido() => print('Guauuuu!');
}

class Gato implements Animal{
  int patas;
  
  void emitirSonido() => print('Miauuuu!');
}
